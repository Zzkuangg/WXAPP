# Room

```
CREATE TABLE `room` (
  `r_id` int(9) NOT NULL,
  `r_name` varchar(9) DEFAULT NULL,
  `r_img` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`r_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
```

# Facility
```
CREATE TABLE `facility` (
  `f_id` varchar(9) NOT NULL,
  `f_name` varchar(20) DEFAULT '',
  `f_room` int(9) DEFAULT NULL,
  `f_img` varchar(100) DEFAULT NULL,
  `f_band` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`f_id`),
  KEY `f_room` (`f_room`),
  CONSTRAINT `facility_ibfk_1` FOREIGN KEY (`f_room`) REFERENCES `room` (`r_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
```
# Cond
```
CREATE TABLE `cond` (
  `c_id` bigint(11) NOT NULL AUTO_INCREMENT,
  `c_facility` varchar(20) DEFAULT NULL,
  `c_state` varchar(20) DEFAULT NULL,
  `c_power` varchar(255) DEFAULT NULL,
  `c_categories` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`c_id`),
  KEY `c_facility` (`c_facility`),
  CONSTRAINT `cond_ibfk_1` FOREIGN KEY (`c_facility`) REFERENCES `facility` (`f_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
```
